# 2. Добавь все файлы

# 3. Сделай первый коммит
git commit -m "first commit"

# 4. Удали старый origin (если есть)

git remote add origin https://github.com/vladimirGoLLUM/Telegram-game-bot.git

# 6. Переименуй ветку в main
git branch -M main

# 7. Отправь на GitHub
git push -u origin main
